214    def update_visualization(self):
215        self.time += self.dt
216        weight: float = 1/len(self.excitation_sources)
217
218        for name, source in self.excitation_sources.items():
219            self.excitation[:] += weight*source(self.time)
220
221            # log.debug(f"Excitation source '{name}' at time {self.time:.3f}s: {source(self.time).shape} shape")
222
223        self.propagator.add_excitation(self.excitation)
224        self.propagator.step()
225        self.Z[:] = self.propagator.get_state()
226
227        Z_vis = cp.asnumpy(self.Z) if self.use_gpu else self.Z
228        max_abs = np.max(np.abs(Z_vis))
229        self.image_item.setLevels([-max_abs, max_abs])
230        self.image_item.setImage(Z_vis, autoLevels=False)
231        self.excitation[:] = 0  # Reset excitation for next step

233class WavePropagatorCPU:
234    def __init__(self, shape, dx, dt, speed, damping):
235        self.shape = shape
236        self.dx = dx
237        self.dt = dt
238        self.c = speed
239        self.damping = damping
240
241        self.Z = np.zeros(shape, dtype=np.float32)
242        self.Z_old = np.zeros_like(self.Z)
243        self.Z_new = np.zeros_like(self.Z)
244
245        self.c2_dt2 = (self.c * self.dt / self.dx)**2
246
247    def add_excitation(self, excitation: np.ndarray):
248        assert excitation.shape == self.Z.shape
249        self.Z += excitation
250
251    def step(self):
252        Z = self.Z
253        laplacian = (
254            -4 * Z +
255            np.roll(Z, 1, axis=0) + np.roll(Z, -1, axis=0) +
256            np.roll(Z, 1, axis=1) + np.roll(Z, -1, axis=1)
257        )
258        self.Z_new = (2 * Z - self.Z_old + self.c2_dt2 * laplacian)
259        self.Z_new *= self.damping
260        self.Z_old = Z.copy()
261        self.Z = self.Z_new.copy()
262
263    def get_state(self):
264        return self.Z
265
266    def reset(self):
267        self.Z[:] = 0
268        self.Z_old[:] = 0
269        self.Z_new[:] = 0
